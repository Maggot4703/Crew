def _save_current_view(self):
    """Save current filtered view to file."""
    try:
        filtered_data = self._get_filtered_data()
        filename = filedialog.asksaveasfilename(
            defaultextension=".csv",
            filetypes=[("CSV files", "*.csv")],
            title="Save filtered view"
        )
        if filename:
            self.file_manager.save_csv_data(filtered_data, filename)
            self.update_status(f"View saved to {filename}")
    except Exception as e:
        self.update_status(f"Error saving view: {e}")

save_button = ttk.Button(button_frame, text="Save View", command=self._save_current_view)
